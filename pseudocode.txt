---STRUKTUR MAHASISWA---

STRUCT Mahasiswa
    nim        : string
    nama       : string
    prodi      : string
    semester   : integer
END STRUCT

DECLARE dataMahasiswa AS list of Mahasiswa


---PROSEDUR TAMBAH DATA (CREATE)---

PROCEDURE TambahData
    DECLARE m AS Mahasiswa

    OUTPUT "Masukkan NIM"
    INPUT m.nim

    OUTPUT "Masukkan Nama"
    INPUT m.nama

    OUTPUT "Masukkan Prodi"
    INPUT m.prodi

    OUTPUT "Masukkan Semester"
    INPUT m.semester

    ADD m TO dataMahasiswa

    OUTPUT "Data berhasil ditambahkan"
END PROCEDURE

---PROSEDUR TAMPILKAN DATA (READ)---

PROCEDURE TampilData
    IF dataMahasiswa is empty THEN
        OUTPUT "Data kosong"
    ELSE
        FOR setiap m IN dataMahasiswa DO
            OUTPUT "NIM      : ", m.nim
            OUTPUT "Nama     : ", m.nama
            OUTPUT "Prodi    : ", m.prodi
            OUTPUT "Semester : ", m.semester
        END FOR
    END IF
END PROCEDURE


--- PROSEDUR UPDATE DATA (UPDATE)---

PROCEDURE UpdateData
    DECLARE cariNIM AS string
    DECLARE ditemukan AS boolean ← false

    OUTPUT "Masukkan NIM yang akan diubah"
    INPUT cariNIM

    FOR setiap m IN dataMahasiswa DO
        IF m.nim = cariNIM THEN
            OUTPUT "Masukkan Nama baru"
            INPUT m.nama

            OUTPUT "Masukkan Prodi baru"
            INPUT m.prodi

            OUTPUT "Masukkan Semester baru"
            INPUT m.semester

            ditemukan ← true
            OUTPUT "Data berhasil diupdate"
            BREAK
        END IF
    END FOR

    IF ditemukan = false THEN
        OUTPUT "Data tidak ditemukan"
    END IF
END PROCEDURE


---PROSEDUR UPDATE DATA (DELETE)---

PROCEDURE HapusData
    DECLARE cariNIM AS string
    DECLARE ditemukan AS boolean ← false

    OUTPUT "Masukkan NIM yang akan dihapus"
    INPUT cariNIM

    FOR i ← 0 TO length(dataMahasiswa)-1 DO
        IF dataMahasiswa[i].nim = cariNIM THEN
            REMOVE dataMahasiswa[i]
            ditemukan ← true
            OUTPUT "Data berhasil dihapus"
            BREAK
        END IF
    END FOR

    IF ditemukan = false THEN
        OUTPUT "Data tidak ditemukan"
    END IF
END PROCEDURE


---MENU UTAMA---

PROCEDURE Menu
    DECLARE pilihan AS integer

    DO
        OUTPUT "===== SISTEM PERKULIAHAN ====="
        OUTPUT "1. Tambah Data Mahasiswa"
        OUTPUT "2. Tampilkan Data Mahasiswa"
        OUTPUT "3. Update Data Mahasiswa"
        OUTPUT "4. Hapus Data Mahasiswa"
        OUTPUT "5. Keluar"

        INPUT pilihan

        SWITCH pilihan
            CASE 1:
                CALL TambahData
            CASE 2:
                CALL TampilData
            CASE 3:
                CALL UpdateData
            CASE 4:
                CALL HapusData
            CASE 5:
                OUTPUT "Terima kasih"
            DEFAULT:
                OUTPUT "Pilihan tidak valid"
        END SWITCH
    WHILE pilihan ≠ 5
END PROCEDURE


---PROGRAM UTAMA---

BEGIN
    CALL Menu
END
